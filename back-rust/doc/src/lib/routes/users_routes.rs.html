use axum::{Router, routing::{get, put, delete}}; // Import des modules nécessaires d'Axum pour créer des routes et gérer les méthodes HTTP (GET, PUT, DELETE).

use crate::users::{get_all_users, update_user, delete_user}; // Import des fonctions liées aux utilisateurs, définies dans le module `users`.

/// Fonction qui définit et retourne les routes liées aux utilisateurs.
pub fn users_routes() -> Router {
    Router::new() // Crée un nouveau routeur pour gérer les routes liées aux utilisateurs.
        .without_v07_checks() // Désactive les vérifications liées à la version 0.7 d'Axum (probablement une configuration liée à un ancien framework ou une version spécifique).
        .route("/{uuid}/update", put(update_user)) // Route pour mettre à jour un utilisateur. Utilise la méthode PUT et appelle la fonction `update_user`. Le `{uuid}` est l'identifiant de l'utilisateur à mettre à jour.
        .route("/{uuid}/delete", delete(delete_user)) // Route pour supprimer un utilisateur. Utilise la méthode DELETE et appelle la fonction `delete_user`. Le `{uuid}` est l'identifiant de l'utilisateur à supprimer.
        .route("/all", get(get_all_users)) // Route pour récupérer tous les utilisateurs. Utilise la méthode GET et appelle la fonction `get_all_users`.
}
