use axum::{Router, routing::{get, post, delete}}; // Importation des modules nécessaires d'Axum pour gérer les routes HTTP.

use crate::assets::{get_assets, get_user_assets, create_asset, delete_asset, get_asset_from_user_and_asset_id}; // Importation des gestionnaires de routes d'assets.


pub fn assets_routes() -> Router {
    Router::new() // Crée un nouveau routeur Axum.
        .route("/", get(get_assets)) // Route pour obtenir tous les assets via une requête GET à la racine ("/").
        .route("/{uuid}", get(get_user_assets)) // Route pour obtenir les assets d'un utilisateur en particulier via son UUID.
        .route("/create", post(create_asset)) // Route pour créer un nouvel asset via une requête POST à "/create".
        .route("/{asset_id}/delete", delete(delete_asset)) // Route pour supprimer un asset particulier en fonction de son ID avec une requête DELETE.
        .route("/{uuid}/{asset_id}/get-asset-from-uuid", get(get_asset_from_user_and_asset_id)) // Route pour obtenir un asset particulier pour un utilisateur donné via leur UUID et l'ID de l'asset.
}
