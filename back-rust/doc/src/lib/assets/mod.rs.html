/// Récupère tous les actifs
///
/// Cette fonction permet de récupérer la liste de tous les actifs (assets) stockés dans la base de données.
/// Elle renvoie la liste des actifs sous forme de JSON.
///
/// # Arguments
/// * `pool` - Le pool de connexions à la base de données pour interagir avec la base de données PostgreSQL.
///
/// # Retour
/// * Un `Result` avec soit la liste des actifs dans un objet JSON, soit une erreur HTTP (status code `500` en cas d'échec).
pub async fn get_assets(
    Extension(pool): Extension<Arc<PgPool>>,
) -> Result<Json<Vec<Asset>>, StatusCode> {
    let assets = sqlx::query_as!(Asset, "SELECT * FROM assets")
        .fetch_all(&*pool)
        .await
        .map_err(|_| StatusCode::INTERNAL_SERVER_ERROR)?;

    Ok(Json(assets))
}

/// Récupère les actifs d'un utilisateur spécifique
///
/// Cette fonction permet de récupérer la liste des actifs (assets) associés à un utilisateur donné, identifié par son `uuid`.
/// Elle renvoie la liste des actifs sous forme de JSON.
///
/// # Arguments
/// * `pool` - Le pool de connexions à la base de données pour interagir avec la base de données PostgreSQL.
/// * `uuid` - L'identifiant de l'utilisateur (UUID) pour lequel récupérer les actifs.
///
/// # Retour
/// * Un `Result` avec soit la liste des actifs dans un objet JSON, soit une erreur HTTP (status code `500` en cas d'échec).
pub async fn get_user_assets(
    Extension(pool): Extension<Arc<PgPool>>,
    Path(uuid): Path<String>,
) -> Result<Json<Vec<Asset>>, StatusCode> {

    let assets = sqlx::query_as!(Asset, r#"SELECT * FROM assets WHERE user_id = $1"#, uuid)
        .fetch_all(&*pool)
        .await
        .map_err(|_| StatusCode::INTERNAL_SERVER_ERROR)?;

    Ok(Json(assets))
}

/// Récupère les actifs d'un utilisateur spécifié par `uuid` et un identifiant d'actif spécifique `nftoken_id`
///
/// Cette fonction permet de récupérer un actif spécifique d'un utilisateur en utilisant l'UUID de l'utilisateur et
/// l'identifiant de l'actif (`nftoken_id`). Elle renvoie l'actif sous forme de JSON.
///
/// # Arguments
/// * `pool` - Le pool de connexions à la base de données pour interagir avec la base de données PostgreSQL.
/// * `uuid` - L'identifiant de l'utilisateur (UUID) pour lequel récupérer l'actif.
/// * `nftoken_id` - L'identifiant de l'actif spécifique à récupérer.
///
/// # Retour
/// * Un `Result` avec soit l'actif demandé dans un objet JSON, soit une erreur HTTP (status code `500` en cas d'échec).
pub async fn get_asset_from_user_and_asset_id(
    Extension(pool): Extension<Arc<PgPool>>,
    Path(uuid): Path<String>,
    Path(nftoken_id): Path<String>,
) -> Result<Json<Vec<Asset>>, StatusCode> {
    let assets = sqlx::query_as!(Asset, r#"SELECT  * FROM assets WHERE user_id = $1 AND nftoken_id = $2"#, uuid, nftoken_id)
        .fetch_all(&*pool)
        .await
        .map_err(|_| StatusCode::INTERNAL_SERVER_ERROR)?;

    Ok(Json(assets))
}

/// Supprime un actif (asset) de la base de données en fonction de son `asset_id`.
///
/// Cette fonction supprime un actif de la base de données à partir de l'identifiant de l'actif
/// fourni dans l'URL de la requête. Elle retourne un code HTTP indiquant si l'opération a réussi ou non.
///
/// # Arguments
/// * `pool` - Le pool de connexions à la base de données pour interagir avec la base de données PostgreSQL.
/// * `asset_id` - L'identifiant de l'actif à supprimer.
///
/// # Retour
/// * Un `Result` contenant un `StatusCode` indiquant le succès ou l'échec de l'opération.
///   - En cas de succès, un code HTTP `204 No Content` est retourné.
pub async fn delete_asset(
    Extension(pool): Extension<Arc<PgPool>>,
    Path(asset_id): Path<String>,
) -> Result<StatusCode, StatusCode> {
    let result = sqlx::query!("DELETE FROM assets WHERE asset_id = $1", asset_id)
        .execute(&*pool)
        .await
        .map_err(|_| StatusCode::INTERNAL_SERVER_ERROR)?;

    match result.rows_affected() {
        0 => Err(StatusCode::NOT_FOUND),
        _ => Ok(StatusCode::NO_CONTENT),
    }
}

/// Télécharge une image dans le répertoire "uploads" avec un nom de fichier au format `asset_id-user_id.extension`
/// Seules les extensions .jpeg et .png sont autorisées.
/// Retourne le chemin du fichier en cas de succès.
///
/// # Arguments
/// * `asset_id` - L'identifiant de l'actif.
/// * `user_id` - L'identifiant de l'utilisateur.
/// * `image_data` - Les données de l'image sous forme de `Vec<u8>`.
/// * `content_type` - Le type de contenu de l'image (par exemple "image/jpeg" ou "image/png").
///
/// # Retour
/// * Un `Result` contenant le chemin du fichier téléchargé en cas de succès,
///   ou un `StatusCode` en cas d'erreur (par exemple, format d'image non pris en charge).
pub async fn upload_image(
    asset_id: String,
    user_id: String,
    image_data: Option<Vec<u8>>,
    content_type: Option<&str>,
) -> Result<String, StatusCode> {
    // Déterminer l'extension du fichier en fonction du type de contenu
    let extension = match content_type {
        Some("image/jpeg") => "jpeg",
        Some("image/png") => "png",
        _ => {
            eprintln!("Unsupported image format: {:?}", content_type);
            return Err(StatusCode::BAD_REQUEST);
        }
    };

    // Créer le répertoire "uploads" s'il n'existe pas
    let upload_dir = PathBuf::from("uploads");
    if !upload_dir.exists() {
        fs::create_dir_all(&upload_dir).map_err(|e| {
            eprintln!("Failed to create uploads directory: {}", e);
            StatusCode::INTERNAL_SERVER_ERROR
        })?;
    }

    let filename = format!("{}-{}.{}", asset_id, user_id, extension);
    let file_path = upload_dir.join(&filename);

    let mut file = File::create(&file_path).await.map_err(|e| {
        eprintln!("Failed to create file: {}", e);
        StatusCode::INTERNAL_SERVER_ERROR
    })?;

    file.write_all(image_data.as_deref().ok_or(StatusCode::BAD_REQUEST)?).await.map_err(|e| {
        eprintln!("Failed to write image data: {}", e);
        StatusCode::INTERNAL_SERVER_ERROR
    })?;

    Ok(file_path.to_string_lossy().into_owned())
}
